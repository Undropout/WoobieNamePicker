extraEmojis[colIndex] : ''; const name = currentMode === 'cosmic-mode' || currentMode === 'prismatic-mode' ? `${adj} ${col} ${ani}\n${extraEmoji}${emoji}` : `${adj} ${col} ${ani} ${emoji}`; if (!currentNames.includes(name)) { const coloration = colorHex[colIndex]; const hexbg = invertColor(coloration); const button = document.createElement('button'); button.className = `name-button ${currentMode}`; button.textContent = name; button.dataset.originalColor = hexbg; button.dataset.originalBg = coloration; button.style.backgroundColor = coloration; button.style.setProperty('--button-bg', coloration); button.onclick = () => selectName(name, hexbg); currentNames.push({name, color: hexbg, bg: coloration}); nameOptions.appendChild(button); } } } function selectName(name, color) { const selectedName = document.getElementById('selected-name'); selectedName.textContent = `Selected: ${name}`; selectedName.style.color = color; } function setMode(mode) { const picker = document.getElementById('name-picker'); const buttons = document.getElementsByClassName('name-button'); picker.className = 'normal-mode'; // Reset to normal picker.classList.add(mode); for (let i = 0; i < buttons.length; i++) { buttons[i].className = `name-button ${mode}`; buttons[i].style.backgroundColor = currentNames[i].bg; buttons[i].style.setProperty('--button-bg', currentNames[i].bg); } currentMode = mode; generateNames(); } document.getElementById('normal-mode').onclick = () => setMode('normal-mode'); document.getElementById('bw-mode').onclick = () => setMode('bw-mode'); document.getElementById('prismatic-mode').onclick = () => setMode('prismatic-mode'); document.getElementById('neon-mode').onclick = () => setMode('neon-mode'); document.getElementById('cosmic-mode').onclick = () => setMode('cosmic-mode'); document.getElementById('reroll-button').onclick = generateNames; generateNames(); // Start in Normal mode
